Order Expansion type Example
1 Brace expansion 
echo foo{A,B,C}bar
2 Tilde expansion 
ls ~/.vimrc
3 Parameter/variable subsitution 
echo ${PATH}
4 Command substitution 
readlink -f $(which python)
5 Arithmetic expansion 
echo $(( 365 * 24 ))
6 Process substitution 
diff <(date) <(sleep 1; date)
7 Word splitting 
args () {
    printf "%d arguments:" "$#"
    printf " <%s>" "$@"
    echo
}

args hello world "how are you?"

VAR="This is a variable"
args $VAR
args "$VAR"
8 Filename expansion 
ls *.pyc
